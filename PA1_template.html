looking for difference in activity patterns btw weekdays and weekends create new factor var in dataset (weekday and weekend)

daytype <- function(date) {
  if (weekdays(as.Date(date)) %in% c("Saturday", "Sunday")) {
    "Weekend"
  } else {
    "Weekday"
  }
}
data$daytype <- as.factor(sapply(data$date, daytype))

create panel plot of 5 min interval and avg # of steps taken, averaged across all weekdays or weekends
